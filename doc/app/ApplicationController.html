def saved_search_query(search_model, cookie_key)
  param = "#{search_model}_search"
  if params[param]
    cookies[cookie_key] = {
      value: params[param].to_json,
      expires: 2.weeks.from_now
    }
  end
  params[param].presence || JSON.load(cookies[cookie_key]) || {}
end
